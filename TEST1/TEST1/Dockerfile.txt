# Stage 1: Build
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copy csproj and restore dependencies
COPY WebHostingTestSite/*.csproj ./WebHostingTestSite/
RUN dotnet restore ./WebHostingTestSite/WebHostingTestSite.csproj

# Copy the rest of the source code
COPY WebHostingTestSite/. ./WebHostingTestSite/

# Build and publish the app
WORKDIR /src/WebHostingTestSite
RUN dotnet publish -c Release -o /app/publish

# Stage 2: Runtime
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS runtime
WORKDIR /app
COPY --from=build /app/publish .

# Expose the default port
EXPOSE 80

# Run the application
ENTRYPOINT ["dotnet", "WebHostingTestSite.dll"]
